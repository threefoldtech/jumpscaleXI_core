```bash
pip3 install -i https://test.pypi.org/simple/ jumpscale11
```

```bash
git clone git@github.com:threefoldtech/jumpscaleXI_core.git
#or:
#git clone https://github.com/threefoldtech/jumpscaleXI_core.git
cd  ~/code/jumpscaleXI_core
pip3 install -e .

```
